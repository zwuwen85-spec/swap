### 2.1 JWT（JSON Web Token）认证

#### Token结构
```
Header.Payload.Signature
```

#### Token Payload
```json
{
  "user_id": 123,
  "username": "zhangsan",
  "role": "user",          // user/admin
  "iat": 1648000000,       // 签发时间
  "exp": 1648604800,       // 过期时间（7天）
  "jti": "unique-id"       // Token唯一标识
}
```

#### Token生成（Go示例）
```go
type Claims struct {
    UserID   int64  `json:"user_id"`
    Username string `json:"username"`
    Role     string `json:"role"`
    jwt.StandardClaims
}

func GenerateToken(userID int64, username string) (string, error) {
    claims := Claims{
        UserID:   userID,
        Username: username,
        Role:     "user",
        StandardClaims: jwt.StandardClaims{
            ExpiresAt: time.Now().Add(7 * 24 * time.Hour).Unix(),
            IssuedAt:  time.Now().Unix(),
            Id:        uuid.New().String(),
        },
    }

    token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
    return token.SignedString([]byte(config.JWTSecret))
}
```

#### Token验证中间件
```go
func AuthMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        authHeader := c.GetHeader("Authorization")
        if authHeader == "" {
            c.JSON(401, gin.H{"code": 10002, "message": "未登录"})
            c.Abort()
            return
        }

        token := strings.TrimPrefix(authHeader, "Bearer ")
        claims, err := ParseToken(token)
        if err != nil {
            c.JSON(401, gin.H{"code": 10002, "message": "Token无效"})
            c.Abort()
            return
        }

        // 检查Token是否在黑名单（用于主动注销）
        if isTokenBlacklisted(claims.Id) {
            c.JSON(401, gin.H{"code": 10002, "message": "Token已失效"})
            c.Abort()
            return
        }

        // 将用户信息存入上下文
        c.Set("user_id", claims.UserID)
        c.Set("username", claims.Username)
        c.Set("role", claims.Role)

        c.Next()
    }
}
```

#### Token刷新机制
```go
// 当Token即将过期时（剩余时间少于1天），自动刷新
func RefreshIfNeeded(token string) (string, error) {
    claims, err := ParseToken(token)
    if err != nil {
        return "", err
    }

    // 剩余时间少于1天，刷新Token
    if claims.ExpiresAt-time.Now().Unix() < 86400 {
        return GenerateToken(claims.UserID, claims.Username)
    }

    return token, nil
}
```

### 2.2 密码安全

#### 密码加密（bcrypt）
```go
import "golang.org/x/crypto/bcrypt"

// 加密密码
func HashPassword(password string) (string, error) {
    // cost=10，约100ms，安全性与性能的平衡
    bytes, err := bcrypt.GenerateFromPassword([]byte(password), 10)
    return string(bytes), err
}

// 验证密码
func CheckPassword(password, hash string) bool {
    err := bcrypt.CompareHashAndPassword([]byte(hash), []byte(password))
    return err == nil
}
```

#### 密码复杂度要求
- 最小长度：6位
- 最大长度：20位
- 必须包含：字母和数字
- 可选：特殊字符

```go
func ValidatePassword(password string) error {
    if len(password) < 6 || len(password) > 20 {
        return errors.New("密码长度必须在6-20位之间")
    }

    // 至少包含字母和数字
    hasLetter := regexp.MustCompile(`[a-zA-Z]`).MatchString(password)
    hasNumber := regexp.MustCompile(`[0-9]`).MatchString(password)

    if !hasLetter || !hasNumber {
        return errors.New("密码必须包含字母和数字")
    }

    return nil
}
```

#### 防止暴力破解
- 登录失败5次后锁定账号30分钟
- 记录失败次数和IP
- 使用验证码（3次失败后）

```go

    return func(c *gin.Context) {
        if c.Request.Header.Get("X-Forwarded-Proto") == "http" {
            httpsURL := "https://" + c.Request.Host + c.Request.URL.String()
```

### 3.3 接口限流

#### 使用Redis实现滑动窗口限流
```go
func RateLimitMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        userID := c.GetInt64("user_id")
        if userID == 0 {
            // 未登录用户：100次/小时
            if !checkRateLimit("ip:"+c.ClientIP(), 100, time.Hour) {
                c.JSON(429, gin.H{"code": 429, "message": "请求过于频繁"})
                c.Abort()
                return
            }
        } else {
            // 已登录用户：300次/小时
            if !checkRateLimit(fmt.Sprintf("user:%d", userID), 300, time.Hour) {
                c.JSON(429, gin.H{"code": 429, "message": "请求过于频繁"})
                c.Abort()
                return
            }
        }

        c.Next()
    }
}

func checkRateLimit(key string, limit int, duration time.Duration) bool {
    count, _ := redis.Incr(key).Result()
    if count == 1 {
        redis.Expire(key, duration)
    }
    return count <= int64(limit)
}
```

### 4.2 敏感数据加密存储

#### 手机号加密（AES）
```go
import (
    "crypto/aes"
    "crypto/cipher"
    "encoding/base64"
)

var aesKey = []byte("your-32-byte-secret-key-1234567890")

func EncryptPhone(phone string) (string, error) {
    block, err := aes.NewCipher(aesKey)
    if err != nil {
        return "", err
    }

    gcm, err := cipher.NewGCM(block)
    if err != nil {
        return "", err
    }

    nonce := make([]byte, gcm.NonceSize())
    ciphertext := gcm.Seal(nonce, nonce, []byte(phone), nil)
    return base64.StdEncoding.EncodeToString(ciphertext), nil
}

func DecryptPhone(encrypted string) (string, error) {
    data, err := base64.StdEncoding.DecodeString(encrypted)
    if err != nil {
        return "", err
    }

    block, err := aes.NewCipher(aesKey)
    if err != nil {
        return "", err
    }

    gcm, err := cipher.NewGCM(block)
    if err != nil {
        return "", err
    }

    nonceSize := gcm.NonceSize()
    nonce, ciphertext := data[:nonceSize], data[nonceSize:]
    plaintext, err := gcm.Open(nil, nonce, ciphertext, nil)
    return string(plaintext), err
}
```

### 4.3 文件上传安全

#### 1. 文件类型验证
```go
func ValidateFile(fileHeader *multipart.FileHeader) error {
    // 检查文件大小（最大5MB）
    if fileHeader.Size > 5*1024*1024 {
        return errors.New("文件大小不能超过5MB")
    }

    // 检查文件扩展名
    ext := strings.ToLower(filepath.Ext(fileHeader.Filename))
    allowedExts := map[string]bool{
        ".jpg":  true,
        ".jpeg": true,
        ".png":  true,
        ".gif":  true,
    }
    if !allowedExts[ext] {
        return errors.New("不支持的文件类型")
    }

    // 检查文件MIME类型（防止伪造扩展名）
    file, err := fileHeader.Open()
    if err != nil {
        return err
    }
    defer file.Close()

    buffer := make([]byte, 512)
    _, err = file.Read(buffer)
    if err != nil {
        return err
    }

    mimeType := http.DetectContentType(buffer)
    allowedMimes := map[string]bool{
        "image/jpeg": true,
        "image/png":  true,
        "image/gif":  true,
    }
    if !allowedMimes[mimeType] {
        return errors.New("不支持的文件类型")
    }

    return nil
}
```

### 5.3 防止重放攻击

#### 请求签名验证
```go
type Request struct {
    Data      string `json:"data"`
    Timestamp int64  `json:"timestamp"`
    Nonce     string `json:"nonce"`
    Signature string `json:"signature"`
}

func VerifySignature(req Request, secret string) bool {
    // 1. 检查时间戳（5分钟内有效）
    if time.Now().Unix()-req.Timestamp > 300 {
        return false
    }

    // 2. 检查Nonce是否已使用（防止重放）
    if redis.Exists("nonce:" + req.Nonce) {
        return false
    }

    // 3. 验证签名
    str := fmt.Sprintf("%s%d%s", req.Data, req.Timestamp, req.Nonce)
    h := hmac.New(sha256.New, []byte(secret))
    h.Write([]byte(str))
    expected := hex.EncodeToString(h.Sum(nil))

    if req.Signature != expected {
        return false
    }

    // 4. 记录Nonce（5分钟过期）
    redis.Set("nonce:"+req.Nonce, 1, 5*time.Minute)

    return true
}
```

```go
func LogOperation(userID int64, module, action, method string, c *gin.Context, status int, err error) {

        Module:    module,
        Action:    action,

    body, _ := c.Get("request_body")
    log.Params = SensitiveDataMask(body)

    db.Create(&log)
```

#### 敏感信息脱敏
```go
func SensitiveDataMask(data interface{}) string {
    str := fmt.Sprintf("%v", data)

    // 手机号脱敏：138****1234
    str = regexp.MustCompile(`1[3-9]\d{9}`).ReplaceAllString(str, "1****$0")

    // 身份证脱敏
    str = regexp.MustCompile(`\d{17}[\dXx]`).ReplaceAllString(str, "***********$0")

    // 密码字段脱敏
    str = regexp.MustCompile(`"password":"[^"]+"`).ReplaceAllString(str, `"password":"***"`)

    return str
}
```

```go
func DetectAbnormalLogin(userID int64, ip string) {
    // 1. 检查是否异地登录（与常用IP差异大）
    commonIPs := redis.SMembers(fmt.Sprintf("user:ips:%d", userID))
    isNewLocation := true
    for _, commonIP := range commonIPs {
        if isSameLocation(ip, commonIP.(string)) {
            isNewLocation = false
            break
        }
    }

    if isNewLocation && len(commonIPs) > 0 {
        // 发送异地登录提醒
        SendSecurityAlert(userID, "检测到异地登录", ip)
    }

    // 2. 记录IP（保留最近10个）
    redis.SAdd(fmt.Sprintf("user:ips:%d", userID), ip)
    redis.Expire(fmt.Sprintf("user:ips:%d", userID), 30*24*time.Hour)
}
```

### 9.2 常见安全事件处理
