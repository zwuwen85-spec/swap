# 校园闲置物品交换平台 - 缓存设计文档

## 一、缓存概述

### 1.1 缓存架构

```
┌─────────────────────────────────────────────────────────┐
│                     应用层                               │
│  ┌─────────┐  ┌─────────┐  ┌─────────┐  ┌─────────┐   │
│  │ Handler │→│  Logic  │→│  Cache  │→│  Model  │   │
│  └─────────┘  └─────────┘  └─────────┘  └─────────┘   │
└─────────────────────────────────────────────────────────┘
                                    ↓
┌─────────────────────────────────────────────────────────┐
│                     Redis集群                            │
│   Master ←→ Slave ←→ Slave                              │
└─────────────────────────────────────────────────────────┘
```

### 1.2 缓存使用场景

| 场景 | 数据类型 | TTL | 说明 |
|-----|---------|-----|------|
| 用户信息 | String | 1h | 登录后缓存 |
| 商品详情 | String | 30m | 查看后缓存 |
| 商品列表 | String | 10m | 首页推荐 |
| 热搜关键词 | ZSet | 1h | 按搜索次数排序 |
| 在线状态 | String | 5m | 心跳更新 |
| 消息未读数 | String | 实时 | 每次登录计算 |
| 分类列表 | String | 1天 | 基本不变 |
| 验证码 | String | 5m | 短期有效 |

### 2.2 用户缓存

#### 2.2.1 用户信息缓存

```go
// Key格式：user:info:{user_id}
// TTL：1小时
// 数据结构：String（JSON）

type UserCache struct {
    redis *redis.Client
}

// 获取用户信息
func (c *UserCache) GetUserInfo(userID int64) (*model.User, error) {
    key := fmt.Sprintf("user:info:%d", userID)

    // 1. 查询缓存
    cached, err := c.redis.Get(key).Result()
    if err == nil {
        var user model.User
        json.Unmarshal([]byte(cached), &user)
        return &user, nil
    }

    // 2. 缓存未命中，查询数据库
    var user model.User
    if err := db.Where("id = ?", userID).First(&user).Error; err != nil {
        return nil, err
    }

    // 3. 写入缓存
    data, _ := json.Marshal(user)
    c.redis.Set(key, data, 1*time.Hour)

    return &user, nil
}

// 更新用户信息（删除缓存）
func (c *UserCache) UpdateUserInfo(user *model.User) error {
    // 1. 更新数据库
    if err := db.Save(user).Error; err != nil {
        return err
    }

    // 2. 删除缓存（而非更新）
    key := fmt.Sprintf("user:info:%d", user.ID)
    c.redis.Del(key)

    return nil
}

// 批量获取用户信息
func (c *UserCache) GetUsersByIDs(userIDs []int64) (map[int64]*model.User, error) {
    keys := make([]string, len(userIDs))
    for i, id := range userIDs {
        keys[i] = fmt.Sprintf("user:info:%d", id)
    }

    // Pipeline批量查询
    pipe := c.redis.Pipeline()
    cmds := make([]*redis.StringCmd, len(keys))
    for i, key := range keys {
        cmds[i] = pipe.Get(key)
    }
    pipe.Exec()

    users := make(map[int64]*model.User)
    missedIDs := make([]int64, 0)

    for i, cmd := range cmds {
        cached, err := cmd.Result()
        if err == nil {
            var user model.User
            json.Unmarshal([]byte(cached), &user)
            users[user.ID] = &user
        } else {
            missedIDs = append(missedIDs, userIDs[i])
        }
    }

    // 查询缓存未命中的用户
    if len(missedIDs) > 0 {
        var dbUsers []model.User
        db.Where("id IN ?", missedIDs).Find(&dbUsers)

        for _, user := range dbUsers {
            users[user.ID] = &user
            // 写入缓存
            data, _ := json.Marshal(user)
            key := fmt.Sprintf("user:info:%d", user.ID)
            c.redis.Set(key, data, 1*time.Hour)
        }
    }

    return users, nil
}
```

```go
type GoodsCache struct {
    redis *redis.Client
}

func (c *GoodsCache) GetGoodsDetail(goodsID int64) (*model.Goods, error) {
    key := fmt.Sprintf("goods:detail:%d", goodsID)

    // 1. 查询缓存
    cached, err := c.redis.Get(key).Result()
    if err == nil {
        var goods model.Goods
        json.Unmarshal([]byte(cached), &goods)
        // 增加浏览次数（异步）
        go c.increaseViewCount(goodsID)
        return &goods, nil
    }

    // 2. 查询数据库
    var goods model.Goods
    if err := db.Where("id = ?", goodsID).First(&goods).Error; err != nil {
        return nil, err
    }

    // 3. 写入缓存
    data, _ := json.Marshal(goods)
    c.redis.Set(key, data, 30*time.Minute)

    return &goods, nil
}

// 增加浏览次数（使用Redis计数器）
func (c *GoodsCache) increaseViewCount(goodsID int64) {
    key := fmt.Sprintf("goods:view:%d", goodsID)
    redis.Incr(key)
    redis.Expire(key, 24*time.Hour)
}

// 批量更新浏览次数到数据库（定时任务）
func SyncViewCountToDB() {
    keys, _ := redis.Keys("goods:view:*").Result()
    for _, key := range keys {
        count, _ := redis.Get(key).Int()
        goodsID := parseGoodsIDFromKey(key)
        db.Exec("UPDATE goods SET view_count = view_count + ? WHERE id = ?", count, goodsID)
        redis.Del(key)
    }
}
```

#### 2.3.2 商品列表缓存

```go
// Key格式：goods:list:{category_id}:{status}:{page}:{filters_hash}
// TTL：10分钟

func GetGoodsList(req *GoodsListRequest) ([]*model.Goods, int64, error) {
    // 1. 生成缓存key
    filtersHash := generateFiltersHash(req)
    key := fmt.Sprintf("goods:list:%d:%d:%d:%s",
        req.CategoryID, req.Status, req.Page, filtersHash)

    // 2. 查询缓存
    cached, err := redis.Get(key).Result()
    if err == nil {
        var result struct {
            List  []*model.Goods `json:"list"`
            Total int64          `json:"total"`
        }
        json.Unmarshal([]byte(cached), &result)
        return result.List, result.Total, nil
    }

    // 3. 查询数据库
    var goods []*model.Goods
    var total int64

    query := db.Model(&model.Goods{}).
        Where("status = ?", req.Status)

    if req.CategoryID > 0 {
        query = query.Where("category_id = ?", req.CategoryID)
    }

    query.Count(&total)
    query.Offset((req.Page - 1) * req.PageSize).
        Limit(req.PageSize).
        Find(&goods)

    // 4. 写入缓存
    result := struct {
        List  []*model.Goods `json:"list"`
        Total int64          `json:"total"`
    }{goods, total}
    data, _ := json.Marshal(result)
    redis.Set(key, data, 10*time.Minute)

    return goods, total, nil
}

// 生成过滤条件的hash
func generateFiltersHash(req *GoodsListRequest) string {
    data := fmt.Sprintf("%d_%d_%s_%s",
        req.Type, req.Condition, req.Sort, req.Keyword)
    h := md5.Sum([]byte(data))
    return hex.EncodeToString(h[:])
}
```

```go
// Key格式：goods:hot:{date}

// 记录商品浏览（用于计算热度）

    today := time.Now().Format("2006-01-02")
    key := fmt.Sprintf("goods:hot:%s", today)

    // 分数+1

    redis.Expire(key, 7*24*time.Hour)

// 获取热门商品TOP10
func GetHotGoods() ([]int64, error) {
    today := time.Now().Format("2006-01-02")
    key := fmt.Sprintf("goods:hot:%s", today)

    // 获取前10名
    results, err := redis.ZRevRangeWithScores(key, 0, 9).Result()
    if err != nil {
        return nil, err
    }

    goodsIDs := make([]int64, len(results))
    for i, result := range results {
        id, _ := strconv.ParseInt(result.Member.(string), 10, 64)
        goodsIDs[i] = id
    }

    return goodsIDs, nil
}
```

```go
func GetUnreadCount(userID int64) (map[int64]int64, error) {
    key := fmt.Sprintf("message:unread:%d", userID)

    result, err := redis.HGetAll(key).Result()
    if err != nil {
        return nil, err
    }

    counts := make(map[int64]int64)

        fid, _ := strconv.ParseInt(friendID, 10, 64)
        c, _ := strconv.ParseInt(count, 10, 64)

    }

    return counts, nil

func IncreaseUnreadCount(userID, friendID int64) {

func ClearUnreadCount(userID, friendID int64) {
    key := fmt.Sprintf("message:unread:%d", userID)
    field := strconv.FormatInt(friendID, 10)
    redis.HDel(key, field)
}
```

### 2.8 分布式锁

```go
// Key格式：lock:{resource}:{id}
// 使用场景：防止重复提交、秒杀等

type DistributedLock struct {
    redis *redis.Client
}

// 获取锁
func (l *DistributedLock) Lock(resource string, id int64, expiration time.Duration) (bool, error) {
    key := fmt.Sprintf("lock:%s:%d", resource, id)
    value := uuid.New().String()

    // SETNX + EXPIRE 原子操作
    ok, err := l.redis.SetNX(context.Background(), key, value, expiration).Result()
    return ok, err
}

// 释放锁
func (l *DistributedLock) Unlock(resource string, id int64) error {
    key := fmt.Sprintf("lock:%s:%d", resource, id)
    return l.redis.Del(context.Background(), key).Err()
}

// 使用示例
func PublishGoods(userID int64, goods *model.Goods) error {
    lock := &DistributedLock{redis: redis}

    // 获取锁（防止重复提交）
    ok, err := lock.Lock("publish", goods.ID, 10*time.Second)
    if err != nil {
        return err
    }
    if !ok {
        return errors.New("操作过于频繁")
    }
    defer lock.Unlock("publish", goods.ID)

    // 业务逻辑
    return db.Create(goods).Error
}
```

```go
// Key格式：rate_limit:{user_id}:{api}
// 使用滑动窗口算法

func RateLimit(userID int64, api string, limit int, duration time.Duration) (bool, error) {
    key := fmt.Sprintf("rate_limit:%d:%s", userID, api)

    // 使用ZSet记录请求时间戳
    now := time.Now().Unix()
    removed, _ := redis.ZRemRangeByScore(context.Background(),
        key, "0", strconv.FormatInt(now-duration.Seconds(), 10)).Result()

    // 获取当前请求数
    count, err := redis.ZCard(context.Background(), key).Result()
    if err != nil {
        return false, err
    }

    // 判断是否超限
    if count >= int64(limit) {
        return false, nil
    }

    // 添加当前请求
    redis.ZAdd(context.Background(), key, &redis.Z{
        Score:  float64(now),
        Member: now,
    })
    redis.Expire(context.Background(), key, duration)

    return true, nil
}
```

## 三、缓存更新策略

### 3.1 Cache-Aside（旁路缓存）

```
读取：
  Client → Cache → 命中 → 返回
                 → 未命中 → DB → Cache → 返回

写入：
  Client → DB → 删除Cache
```

**优点：**
- 简单易实现
- 缓存失效由DB写入触发

**缺点：**
- 首次读取必然miss
- 需要处理并发问题

### 3.3 缓存击穿解决方案

**方案1：互斥锁**

```go
func GetGoods(goodsID int64) (*model.Goods, error) {
    key := fmt.Sprintf("goods:detail:%d", goodsID)

    // 1. 查询缓存
    cached, err := redis.Get(key).Result()
    if err == nil {
        var goods model.Goods
        json.Unmarshal([]byte(cached), &goods)
        return &goods, nil
    }

    // 2. 获取分布式锁
    lockKey := fmt.Sprintf("lock:goods:%d", goodsID)
    ok, _ := redis.SetNX(context.Background(),
        lockKey, 1, 10*time.Second).Result()
    if !ok {
        // 未获取到锁，等待100ms后重试
        time.Sleep(100 * time.Millisecond)
        return GetGoods(goodsID)
    }
    defer redis.Del(context.Background(), lockKey)

    // 3. 双重检查（可能其他线程已经加载了）
    cached, _ = redis.Get(key).Result()
    if cached != "" {
        var goods model.Goods
        json.Unmarshal([]byte(cached), &goods)
        return &goods, nil
    }

    // 4. 查询数据库
    var goods model.Goods
    if err := db.Where("id = ?", goodsID).First(&goods).Error; err != nil {
        return nil, err
    }

    // 5. 写入缓存
    data, _ := json.Marshal(goods)
    redis.Set(key, data, 30*time.Minute)

    return &goods, nil
}
```

```go
// L1: 本地缓存
var localCache = &sync.Map{}

// L2: Redis缓存
func GetGoodsWithL2(goodsID int64) (*model.Goods, error) {
    // 1. 本地缓存
    if cached, ok := localCache.Load(goodsID); ok {
        return cached.(*model.Goods), nil
    }

    // 2. Redis缓存
    key := fmt.Sprintf("goods:detail:%d", goodsID)
    cached, err := redis.Get(key).Result()
    if err == nil {
        var goods model.Goods
        json.Unmarshal([]byte(cached), &goods)

        return &goods, nil

    var goods model.Goods
    if err := db.Where("id = ?", goodsID).First(&goods).Error; err != nil {
        return nil, err
    }

    // 回写缓存
    data, _ := json.Marshal(goods)
    redis.Set(key, data, 30*time.Minute)
    localCache.Store(goodsID, &goods)

    return &goods, nil
}
```

## 四、缓存监控

### 4.1 缓存命中率监控

```go
type CacheStats struct {
    Hit   int64
    Miss  int64
}

var stats = &CacheStats{}

func GetFromCache(key string) (string, error) {
    val, err := redis.Get(key).Result()
    if err == nil {
        atomic.AddInt64(&stats.Hit, 1)
        return val, nil
    }

    atomic.AddInt64(&stats.Miss, 1)
    return "", err
}

func GetHitRate() float64 {
    total := atomic.LoadInt64(&stats.Hit) + atomic.LoadInt64(&stats.Miss)
    if total == 0 {
        return 0
    }
    return float64(atomic.LoadInt64(&stats.Hit)) / float64(total)
}
```

## 五、最佳实践

### 5.1 缓存最佳实践

| 场景 | 策略 |
|-----|------|
| 读多写少 | Cache-Aside |
| 读写一致性强 | Write-Through |
| 允许短时不一致 | Write-Behind |
| 热点数据 | 永不过期+后台更新 |
| 计数类 | Redis计数+定时落库 |

### 5.2 注意事项

1. **避免大key**：单个value不超过1MB
2. **避免热key**：使用分片或本地缓存
3. **设置合理TTL**：根据业务需求
4. **监控缓存命中率**：低于80%需要优化
5. **做好容错**：缓存故障不影响业务
6. **及时删除无用key**：避免内存泄漏
